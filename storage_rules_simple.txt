rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // ВРЕМЕННЫЕ ПРАВИЛА ДЛЯ ТЕСТИРОВАНИЯ
    // Разрешить всем аутентифицированным пользователям (включая анонимных)
    match /{allPaths=**} {
      allow read, write: if request.auth != null && 
                            request.resource != null && 
                            request.resource.size < 10 * 1024 * 1024; // Max 10MB
    }
  }
}
